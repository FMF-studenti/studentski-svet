\section{Splošne odločbe}

\begin{clen}[sestava]
  Komisija za študentska mnenja \vnadalj{komisija ali KŠM} je organ Študentskega
  sveta FMF \vnadalj{ŠS FMF}. Sestavljajo jo trije do štirje študenti Fakultete
  za matematiko in fiziko \vnadalj{FMF}, ki niso nujno člani ŠS FMF.
  Zaželeno je, da je vsak kandidat iz ene smeri študija MAT (VŠ in UN) ter FIZ
  (VŠ in UN). Komisijo imenuje in potrdi ŠS FMF na konstitutivni seji, ko
  imenuje tudi predsednika komisije, ki pa mora biti član ŠS FMF.

  Članstvo v komisiji je tajno. Niti ŠS FMF niti komisija ne sme izdati identitete
  članov komisije ali storiti karkoli, kar bi lahko ogrozilo njihovo anonimnost.
\end{clen}

\begin{clen}[pravna osnova]
  Delovanje komisije ureja Poslovnik KŠM.
  Poslovnik začne veljati s \today, ko ga potrdita ŠS FMF in komisija.
\end{clen}

\begin{clen}[mandat]
  Mandat komisije traja do imenovanja nove komisije oziroma največ eno leto.
\end{clen}

\begin{clen}[dolžnosti]
  Namen komisije je, da na podlagi študentskih anket in mnenj študentov oceni
  pedagoško delo predavatelja ter napiše ustrezno mnenje na prošnjo senata FMF. Če
  predavatelj potrebuje mnenje za izvolitev v naziv, komisija v skladu s tem
  poslovnikom presodi, ali se z izvolitvijo strinja in iz pozitivno ali
  negativno mnenje.

  Komisija je dolžna obravnavati vse prošnje za sestavo študentskih mnenj, ki
  jih posreduje ŠS FMF.
\end{clen}

\section{Sestava študentskih mnenj}
\begin{clen}[zbiranje podatkov]
  \label{clen:zbiranje}
  Ko komisija prejme prošnjo, začne z zbiranjem rezultatov študentskih anket.
  To vključuje študentske ankete, ki jih izvaja univerza (univerzitetne ankete),
  FMF (interne ankete) ter podatke o predhodnih ocenjevanjih predavatelja glede na
  pretekla mnenja komisije. Komisija lahko povpraša za mnenje tudi študente, ki
  jim predavatelj trenutno predava in sprejema anonimna mnenja v predalčku ŠS
  FMF in na naslov
  \href{mailto:studentski-svet@list.fmf.uni-lj.si}{studentski-svet@list.fmf.uni-lj.si}.
\end{clen}

\begin{clen}[sestava mnenja]
  Ko komisija zbere vse podatke se z njimi seznanijo vsi člani komisije. Nato
  komisija diskutira o pedagoškem delu predavatelja glede na zbrane podatke in
  kriterije, omenjene v nadaljnjem besedilu.

  Po diskusiji komisija odloči med dvema možnostma; poda bodisi pozitivno
  mnenje, bodisi negativno. O odločitvi komisija odloča z glasovanjem. Za
  sprejem so potrebni vsi glasovi komisije.

  Komisija nato poda pisno mnenje in ga podkrepi z navedbo podatkov, ki jih je
  zbrala. Poskusi podariti prednosti in šibke točke pedagoškega delavca, z
  namenom izboljšanja kvalitete študijskega procesa.  Še posebej natančno mora
  navesti razloge za negativno mnenje, da s tem spodbudi kandidata k spremembam
  v procesu dela.
\end{clen}

\begin{clen}[predlog mnenja]
  Predsednik komisije poroča ŠS FMF o predlaganih mnenjih, ki so obravnavana
  na seji ŠS FMF.
\end{clen}

\section{Kriteriji za sestavo študentskega mnenja}

\begin{clen}[objektivnost in neodvisnost]
  Komisija si mora prizadevati, da pedagoško delo predavatelja oceni čimbolj
  objektivno, nepristransko ter da zastopa mnenje cele študentske skupnosti pred
  mnenjem posameznika.

  Na oceno komisije ne sme vplivati nihče, ki ni član komisije.  V nasprotnem
  primeru lahko komisija poda pritožbo ŠS FMF, ki kršitev ustrezno obravnava.
\end{clen}

\begin{clen}[ustreznost kandidatov]
  Če predavatelj še ni opravljal pedagoškega dela, komisija mnenja ne poda.
  Anketa se pri oceni upošteva, če je nanjo odgovorilo najmanj 8 študentov. V
  nasprotnem primeru se upošteva mnenje posameznih študentov.
\end{clen}

\begin{clen}[splošni kriteriji]
  Komisija na podlagi podatkov iz~\ref{clen:zbiranje} presodi o svoji odločitvi.
  Pri tem imajo univerzitetne ankete prednost pred internimi, interne pa pred
  mnenji posameznih študentov. Če ima predavatelj pri katerikoli univerzitetni
  anketi končni rezultat negativen, komisija lahko poda negativno mnenje. Če ima
  kandidat vse končne rezultate univerzitetnih višje od 2, komisija poda
  pozitivno mnenje.

  Če kandidat prosi za naziv rednega profesorja in je skupen rezultat katerekoli
  merodajne ankete negativen, potem komisija poda negativno mnenje.
\end{clen}

\begin{clen}[ponavljanje zadržkov]
  Če komisija dobi v obravnavo kandidata, za katerega ima pri kateri izmed
  prejšnjih obravnav zapisan kakšen izrecen zadržek, ki se do tekoče obravnave
  ni popravil, potem komisija poda negativno mnenje.
\end{clen}
